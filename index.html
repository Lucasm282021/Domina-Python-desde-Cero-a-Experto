class DesarrolladorPython:
    def __init__(self, nombre):
        self.nombre = nombre
        self.skills = ['Python Core']

    def aprender(self, modulo):
        self.skills.append(modulo)
        print(f"¡{self.nombre} ha dominado {modulo}!")

estudiante = DesarrolladorPython("Tú")
estudiante.aprender("Desarrollo Web con Django")
# Output: ¡Tú has dominado Desarrollo Web con Django!
